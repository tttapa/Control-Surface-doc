98 #ifdef DEBUG_MIDI_PACKETS
99  // TODO: print cable
100  if (midimsg.hasTwoDataBytes())
101  DEBUG(">>> " << hex << midimsg.header << ' ' << midimsg.data1 << ' '
102  << midimsg.data2 << dec);
103  else
104  DEBUG(">>> " << hex << midimsg.header << ' ' << midimsg.data1 << dec);
105 #endif
106 
107  // If the Channel Message callback exists, call it to see if we have to
108  // continue handling it.
109  if (channelMessageCallback && channelMessageCallback(midimsg))
110  return;
111 
112  if (midimsg.getMessageType() == MIDIMessageType::CONTROL_CHANGE &&
113  midimsg.getData1() == MIDI_CC::Reset_All_Controllers) {
114  // Reset All Controllers
115  DEBUG(F("Reset All Controllers"));
116  MIDIInputElementCC::resetAll();
117  MIDIInputElementCP::resetAll();
118  } else if (midimsg.getMessageType() == MIDIMessageType::CONTROL_CHANGE &&
119  midimsg.getData1() == MIDI_CC::All_Notes_Off) {
120  // All Notes Off
121  MIDIInputElementNote::resetAll();
122  } else {
123  switch (midimsg.getMessageType()) {
124  case MIDIMessageType::NOTE_OFF: // fallthrough
125  case MIDIMessageType::NOTE_ON:
126  DEBUGFN(F("Updating Note elements with new MIDI "
127  "message."));
128  MIDIInputElementNote::updateAllWith(midimsg);
129  break;
130  case MIDIMessageType::KEY_PRESSURE:
131  DEBUGFN(F("Updating Key Pressure elements with new MIDI "
132  "message."));
133  MIDIInputElementKP::updateAllWith(midimsg);
134  break;
135  case MIDIMessageType::CONTROL_CHANGE:
136  DEBUGFN(F("Updating CC elements with new MIDI "
137  "message."));
138  MIDIInputElementCC::updateAllWith(midimsg);
139  break;
140  case MIDIMessageType::PROGRAM_CHANGE:
141  DEBUGFN(F("Updating Program Change elements with new MIDI "
142  "message."));
143  MIDIInputElementPC::updateAllWith(midimsg);
144  break;
145  case MIDIMessageType::CHANNEL_PRESSURE:
146  DEBUGFN(F("Updating Channel Pressure elements with new MIDI "
147  "message."));
148  MIDIInputElementCP::updateAllWith(midimsg);
149  break;
150  case MIDIMessageType::PITCH_BEND:
151  // Channel Pressure
152  DEBUGFN(F("Updating Pitch Bend elements with new MIDI "
153  "message."));
154  MIDIInputElementPB::updateAllWith(midimsg);
155  break;
156  // These MIDI types are not channel messages, so aren't handled here
157  case MIDIMessageType::SYSEX_START: break; // LCOV_EXCL_LINE
158  case MIDIMessageType::SYSEX_END: break; // LCOV_EXCL_LINE
159  case MIDIMessageType::TUNE_REQUEST: break; // LCOV_EXCL_LINE
160  case MIDIMessageType::TIMING_CLOCK: break; // LCOV_EXCL_LINE
161  case MIDIMessageType::UNDEFINED_REALTIME_1: break; // LCOV_EXCL_LINE
162  case MIDIMessageType::START: break; // LCOV_EXCL_LINE
163  case MIDIMessageType::CONTINUE: break; // LCOV_EXCL_LINE
164  case MIDIMessageType::STOP: break; // LCOV_EXCL_LINE
165  case MIDIMessageType::UNDEFINED_REALTIME_2: break; // LCOV_EXCL_LINE
166  case MIDIMessageType::ACTIVE_SENSING: break; // LCOV_EXCL_LINE
167  case MIDIMessageType::RESET: break; // LCOV_EXCL_LINE
168  default: break; // LCOV_EXCL_LINE
169  }
170  }

209 void Control_Surface_::updateDisplays() {
210  auto &allElements = DisplayElement::getAll();
211  auto it = allElements.begin();
212  auto end = allElements.end();
213  if (it == end)
214  return;
215  auto prevIt = it;
216  auto previousDisplay = &prevIt->getDisplay();
217  bool dirty = false;
218  // Loop over all display elements
219  while (true) {
220  dirty |= (*it).getDirty();
221  ++it;
222  // If this is the first element on another display
223  if (it == end || &it->getDisplay() != previousDisplay) {
224  // If there was at least one element on the previous display that
225  // has to be redrawn
226  if (dirty) {
227  // Clear the display
228  previousDisplay->clearAndDrawBackground();
229  // Update all elements on that display
230  for (auto drawIt = prevIt; drawIt != it; ++drawIt)
231  drawIt->draw();
232  // Write the buffer to the display
233  previousDisplay->display();
234  }
235  if (it == end)
236  break;
237  prevIt = it;
238  previousDisplay = &it->getDisplay();
239  dirty = false;
240  }
241  }
