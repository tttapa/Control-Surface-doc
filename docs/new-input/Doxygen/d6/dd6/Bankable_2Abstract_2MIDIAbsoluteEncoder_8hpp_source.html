39  Enc_t encval = encoder.read();
40  // If Enc_t is an unsigned type, integer overflow is well-defined, which
41  // is what we want when Enc_t is small and expected to overflow.
42  // However, we need it to be signed because we're interested in the
43  // delta.
44  Enc_t uDelta = encval - deltaOffset;
45  if (uDelta) {
46  int16_t delta = SignedEnc_t(uDelta);
47  // Assumption: delta and speedMultiply are relatively small, so
48  // multiplication probably won't overflow.
49  int16_t multipliedDelta = delta * speedMultiply + remainder;
50  int16_t scaledDelta = multipliedDelta / pulsesPerStep;
51  remainder = multipliedDelta % pulsesPerStep;
52 
53  address.lock();
54  int16_t oldValue = values[address.getSelection()];
55  int16_t newValue = oldValue + scaledDelta;
56  newValue = constrain(newValue, 0, maxValue);
57  if (oldValue != newValue) {
58  sender.send(newValue, address.getActiveAddress());
59  values[address.getSelection()] = newValue;
60  }
61  address.unlock();
62

64  }
