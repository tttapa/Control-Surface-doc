116 void MIDI_Interface::updateIncoming(MIDIInterface_t *iface) {
117  if (iface->getStaller() == iface)
118  iface->unstall(iface);
119  bool chunked = false;
120  MIDIReadEvent event = iface->read();
121  while (event != MIDIReadEvent::NO_MESSAGE) {
122  dispatchIncoming(iface, event);
123  if (event == MIDIReadEvent::SYSEX_CHUNK)
124  chunked = true;
125  if (event == MIDIReadEvent::SYSEX_MESSAGE)
126  chunked = false;
127  event = iface->read();
128  }
129  if (chunked)
130  iface->stall(iface);
131  // TODO: add logic to detect MIDI messages such as (N)RPN that span over
132  // multiple channel voice messages and that shouldn't be interrupted.
133  // For short messages such as (N)RPN, I suggest waiting with a timeout.
134 }

157 void MIDI_Interface::handleStall(MIDIInterface_t *iface) {
158  iface->unstall(iface);
159 
160  unsigned long startTime = millis();
161  while (millis() - startTime < SYSEX_CHUNK_TIMEOUT) {
162  MIDIReadEvent event = iface->read();
163  dispatchIncoming(iface, event);
164  if (event == MIDIReadEvent::SYSEX_CHUNK)
165  startTime = millis(); // reset timeout
166  else if (event == MIDIReadEvent::SYSEX_MESSAGE)
167  return;
168  }
169  DEBUGREF(F("Warning: Unable to un-stall pipes: ")
