24  // First swap the normal member variables:
25  enc_util::swap(this->pin1, other.pin1);
26  enc_util::swap(this->pin2, other.pin2);
27  enc_util::swap(this->interrupts_in_use, other.interrupts_in_use);

29  // Next, update the pointers in interruptArgs:
30  // When interrupts are in use, there is a global interrupt context
31  // variable that holds a pointer to the encoders that are being swapped
32  // or moved.
33  // After moving, this pointer would no longer be valid, so it has to be
34  // changed to point to the new encoder object.
35  // Calling attachInterrupt is not necessary, because this should already
36  // have happened in the begin method if interrupts_in_use is nonzero.
37  // Before messing with the state variables that can be changed or
38  // accessed from within an ISR, we have to disable interrupts.
39  noInterrupts();
