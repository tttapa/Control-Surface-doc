156 uint16_t BLEMIDI::getMinMTU() const {
157  if (pServer == nullptr) {
158  ERROR(F("Error: Server has not been initialized"), 0x2320);
159  return 23; // Set to maximum and hope for the best
160  }
161  auto peers = pServer->getPeerDevices(false);
162  using value_t = decltype(peers)::value_type;
163  auto compare = [](const value_t &a, const value_t &b) {
164  return a.second.mtu < b.second.mtu;
165  };
166  auto min = std::min_element(std::begin(peers), std::end(peers), compare);
167  if (min != std::end(peers))
168  return min->second.mtu;
169  return 23;
170 }
