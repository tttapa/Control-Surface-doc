138 MIDIReadEvent USBMIDI_Parser::feed(MIDIUSBPacket_t packet) {
139  // DEBUG("MIDIUSB packet:\t" << hex << packet[0] << ' ' << packet[1] << ' '
140  // << packet[2] << ' ' << packet[3] << dec);
141 
142  // MIDI USB cable number and code index number
143  Cable cable = Cable(packet[0] >> 4);
144  MIDICodeIndexNumber CIN = MIDICodeIndexNumber(packet[0] & 0xF);
145 
146  // Ignore all messages for cables that we don't have
147  if (cable.getRaw() >= USB_MIDI_NUMBER_OF_CABLES)
148  return MIDIReadEvent::NO_MESSAGE; // LCOV_EXCL_LINE
149 
150  switch (CIN) {
151  case MIDICodeIndexNumber::MISC_FUNCTION_CODES: break; // LCOV_EXCL_LINE
152  case MIDICodeIndexNumber::CABLE_EVENTS: break; // LCOV_EXCL_LINE
153 
154  case MIDICodeIndexNumber::SYSTEM_COMMON_2B: // fallthrough
155  case MIDICodeIndexNumber::SYSTEM_COMMON_3B:
156  return handleSysCommon(packet, cable);
157 
158  case MIDICodeIndexNumber::SYSEX_START_CONT:
159  return handleSysExStartContinue(packet, cable);
160 
161  case MIDICodeIndexNumber::SYSEX_END_1B:
162  return handleSysExEnd1B(packet, cable);
163  case MIDICodeIndexNumber::SYSEX_END_2B:
164  return handleSysExEnd<2>(packet, cable);
165  case MIDICodeIndexNumber::SYSEX_END_3B:
166  return handleSysExEnd<3>(packet, cable);
167 
168  case MIDICodeIndexNumber::NOTE_OFF: // fallthrough
169  case MIDICodeIndexNumber::NOTE_ON: // fallthrough
170  case MIDICodeIndexNumber::KEY_PRESSURE: // fallthrough
171  case MIDICodeIndexNumber::CONTROL_CHANGE: // fallthrough
172  case MIDICodeIndexNumber::PROGRAM_CHANGE: // fallthrough
173  case MIDICodeIndexNumber::CHANNEL_PRESSURE: // fallthrough
174  case MIDICodeIndexNumber::PITCH_BEND:
175  return handleChannelMessage(packet, cable);
176 
177  case MIDICodeIndexNumber::SINGLE_BYTE:
178  return handleSingleByte(packet, cable);
179 
180  default: break; // LCOV_EXCL_LINE
181  }
182 
183  return MIDIReadEvent::NO_MESSAGE; // LCOV_EXCL_LINE
