121 MIDIReadEvent USBMIDI_Parser::feed(MIDIUSBPacket_t packet) {
122  // DEBUG("MIDIUSB packet:\t" << hex << packet[0] << ' ' << packet[1] << ' '
123  // << packet[2] << ' ' << packet[3] << dec);
124 
125  // MIDI USB cable number and code index number
126  Cable cable = Cable(packet[0] >> 4);
127  MIDICodeIndexNumber CIN = MIDICodeIndexNumber(packet[0] & 0xF);
128 
129  // Ignore all messages for cables that we don't have
130  if (cable.getRaw() >= USB_MIDI_NUMBER_OF_CABLES)
131  return MIDIReadEvent::NO_MESSAGE; // LCOV_EXCL_LINE
132 
133  switch (CIN) {
134  case MIDICodeIndexNumber::MISC_FUNCTION_CODES: break; // LCOV_EXCL_LINE
135  case MIDICodeIndexNumber::CABLE_EVENTS: break; // LCOV_EXCL_LINE
136 
137  case MIDICodeIndexNumber::SYSTEM_COMMON_2B: // fallthrough
138  case MIDICodeIndexNumber::SYSTEM_COMMON_3B:
139  return handleSysCommon(packet, cable);
140 
141  case MIDICodeIndexNumber::SYSEX_START_CONT:
142  return handleSysExStartContinue(packet, cable);
143 
144  case MIDICodeIndexNumber::SYSEX_END_1B:
145  return handleSysExEnd1B(packet, cable);
146  case MIDICodeIndexNumber::SYSEX_END_2B:
147  return handleSysExEnd<2>(packet, cable);
148  case MIDICodeIndexNumber::SYSEX_END_3B:
149  return handleSysExEnd<3>(packet, cable);
150 
151  case MIDICodeIndexNumber::NOTE_OFF: // fallthrough
152  case MIDICodeIndexNumber::NOTE_ON: // fallthrough
153  case MIDICodeIndexNumber::KEY_PRESSURE: // fallthrough
154  case MIDICodeIndexNumber::CONTROL_CHANGE: // fallthrough
155  case MIDICodeIndexNumber::PROGRAM_CHANGE: // fallthrough
156  case MIDICodeIndexNumber::CHANNEL_PRESSURE: // fallthrough
157  case MIDICodeIndexNumber::PITCH_BEND:
158  return handleChannelMessage(packet, cable);
159 
160  case MIDICodeIndexNumber::SINGLE_BYTE:
161  return handleSingleByte(packet, cable);
162 
163  default: break; // LCOV_EXCL_LINE
164  }
165 
166  return MIDIReadEvent::NO_MESSAGE; // LCOV_EXCL_LINE
