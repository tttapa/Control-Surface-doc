24 void BluetoothMIDI_Interface::handleSendEvents() {
25  lock_t lock(mtx);
26 
27  // Wait for a packet to be started (or for a stop signal)
28  cv.wait(lock, [this] { return !packetbuilder.empty() || stop_sending; });
29  // Wait for flush signal or timeout
30  bool flushing = cv.wait_for(lock, timeout, [this] { return flushnow; });
31 
32  // Send the packet over BLE, empty the buffer, and update the buffer size
33  // based on the MTU of the connected clients.
34  bleMidi.notifyValue(packetbuilder.getBuffer(), packetbuilder.getSize());
35  packetbuilder.reset();
36  packetbuilder.setCapacity(min_mtu - 3);
37 
38  // Notify the main thread that the flush was done
39  if (flushing) {
40  flushnow = false;
41  lock.unlock();
42  cv.notify_one();
43  }
44 }

46 void BluetoothMIDI_Interface::flushImpl(lock_t &lock) {
47  assert(lock.owns_lock());
48  // No need to send empty packets
49  if (packetbuilder.empty())
50  return;

52  // Tell the background sender thread to send the packet now
53  flushnow = true;

55  cv.notify_one();
56

58  lock.lock();

86 void BluetoothMIDI_Interface::sendChannelMessageImpl3Bytes(ChannelMessage msg) {
87  // BLE packets are sent asynchronously, so we need a lock to access the
88  // packet buffer
89  lock_t lock(mtx);
90  uint16_t timestamp = millis();
91  // Try adding the message to the current packet
92  if (!packetbuilder.add3B(msg.header, msg.data1, msg.data2, timestamp)) {
93  // If that doesn't work, flush the packet (send it now and wait until
94  // it is sent)
95  flushImpl(lock);
96  // And then add it to the (now emtpy) buffer
97  packetbuilder.add3B(msg.header, msg.data1, msg.data2, timestamp);
98  }
99  // Notify the packet sender that data has been added to the buffer
100  lock.unlock();
101  cv.notify_one();
102 }

130 void BluetoothMIDI_Interface::sendSysExImpl(SysExMessage msg) {
131  lock_t lock(mtx);
132 
133  size_t length = msg.length;
134  const uint8_t *data = msg.data;
135  uint16_t timestamp = millis(); // BLE MIDI timestamp
136 
137  // Try adding at least the SysExStart header to the current packet
138  if (!packetbuilder.addSysEx(data, length, timestamp)) {
139  // If that didn't fit, flush the packet
140  flushImpl(lock);
141  // Add the first part of the SysEx message to this packet
142  packetbuilder.addSysEx(data, length, timestamp);
143  }
144  // As long as there's data to be sent in the next packet
145  while (data) {
146  // Send the previous (full) packet
147  flushImpl(lock);
148  // And add the next part of the SysEx message to a continuation packet
149  packetbuilder.continueSysEx(data, length, timestamp);
150  }
151  // Notify the packet sender that data has been added to the buffer
152  lock.unlock();
153  cv.notify_one();
154 }
