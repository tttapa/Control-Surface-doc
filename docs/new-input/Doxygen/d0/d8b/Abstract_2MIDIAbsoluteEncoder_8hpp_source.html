37  // If Enc_t is an unsigned type, integer overflow is well-defined, which
38  // is what we want when Enc_t is small and expected to overflow.
39  // However, we need it to be signed because we're interested in the

41  Enc_t uDelta = encval - deltaOffset;
42  if (uDelta) {

44  // Assumption: delta and speedMultiply are relatively small, so
45  // multiplication probably won't overflow.
46  int16_t multipliedDelta = delta * speedMultiply + remainder;
47  int16_t scaledDelta = multipliedDelta / pulsesPerStep;
48  remainder = multipliedDelta % pulsesPerStep;
