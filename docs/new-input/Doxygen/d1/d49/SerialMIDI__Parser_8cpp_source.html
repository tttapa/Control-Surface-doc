11  // If it's a Real-Time message
12  if (midiByte >= uint8_t(MIDIMessageType::TIMING_CLOCK)) {
13  return handleRealTime(midiByte);
14  }
15 
16  // Normal header (channel message, system exclusive, system common)
17  else {
18  // If a SysEx message was being received, and now we receive
19  // another status byte, remember to correctly terminate the SysEx
20  // message later
21  bool unterminatedSysEx =
22  midimsg.header == uint8_t(MIDIMessageType::SYSEX_START);
23 
24  // Tune Request is a special System Common message of 1 byte
25  if (midiByte == uint8_t(MIDIMessageType::TUNE_REQUEST)) {
26  // Tune request (not implemented)
27  // TODO: should I implement this as a Real-Time message?
28  // (That might lead to problems when a SysEx message is
29  // terminated by a Tune Request, see unterminatedSysEx below)
30  // TODO: should I set midimsg.header in this case?
31  if (unterminatedSysEx) {
32  // If we were previously receiving a SysEx message, this Tune
33  // Request should terminate that SysEx message first.
34  // We cannot return both the SysEx message and the Tune Request
35  // at the same time, so store the Tune Request for later.
36  storeByte(midiByte);
37  } else {
38  midimsg.header = midiByte;
39  midimsg.data1 = 0;
40  midimsg.data2 = 0;
41  return MIDIReadEvent::SYSCOMMON_MESSAGE;
42  // TODO: clear running status
43  }
44  }

48  // the message (if buffer size allows it)

51  if (!hasSysExSpace()) {

98 MIDIReadEvent SerialMIDI_Parser::handleData(uint8_t midiByte) {
99  // If this is the third byte of three (second data byte)
100  if (thirdByte) {
101  // If it's a channel message
102  if (midimsg.hasValidChannelMessageHeader()) {
103  midimsg.data2 = midiByte;
104  // Next byte is either a header or the first data byte of the next
105  // message, so clear the thirdByte flag
106  thirdByte = false;
107  return MIDIReadEvent::CHANNEL_MESSAGE;
108  }
109  // If it's a system common message
110  else if (midimsg.hasValidSystemCommonHeader()) {
111  midimsg.data2 = midiByte;
112  thirdByte = false;
113  return MIDIReadEvent::SYSCOMMON_MESSAGE;
114  } else {
115  ERROR(F("Unexpected third byte"), 0x3333); // LCOV_EXCL_LINE
116  return MIDIReadEvent::NO_MESSAGE; // LCOV_EXCL_LINE
117  }
118  }
119 
120  // If this is not the third byte of three, it's either the second byte
121  // (first data byte) of a channel message, or a SysEx data byte
122 
123  // If the previous SysEx was complete, but no other headers were received
124  else if (midimsg.header == uint8_t(MIDIMessageType::SYSEX_END)) {
125  DEBUGREF(F("Unexpected data after SysExEnd"));
126  return MIDIReadEvent::NO_MESSAGE;
127  }
128 
129  // If it's a channel message
130  else if (midimsg.hasValidChannelMessageHeader()) {
131  // If it's a channel message with two data bytes
132  if (ChannelMessage(midimsg).hasTwoDataBytes()) {
133  midimsg.data1 = midiByte;
134  // We've received the second byte, expect the third byte next
135  thirdByte = true;
136  return MIDIReadEvent::NO_MESSAGE;
137  }
138  // If it's a channel message with one data byte
139  else {
140  midimsg.data1 = midiByte;
141  midimsg.data2 = 0;
142  // The message is finished
143  return MIDIReadEvent::CHANNEL_MESSAGE;
144  }
145  }
146 
147  // If it's a system common message
148  else if (midimsg.hasValidSystemCommonHeader()) {
149  // If it's a channel message with two data bytes
150  if (SysCommonMessage(midimsg).getNumberOfDataBytes() == 2) {
151  midimsg.data1 = midiByte;
152  // We've received the second byte, expect the third byte next
153  thirdByte = true;
154  return MIDIReadEvent::NO_MESSAGE;
155  }
156  // If it's a channel message with one data byte
157  else if (SysCommonMessage(midimsg).getNumberOfDataBytes() == 1) {
158  midimsg.data1 = midiByte;
159  midimsg.data2 = 0;
160  // The message is finished
161  return MIDIReadEvent::SYSCOMMON_MESSAGE;
162  } else {
163  DEBUGREF(F("Unexpected data byte"));
164  return MIDIReadEvent::NO_MESSAGE;
165  }
166  // TODO: clear running status
167  }
168 
169  // Otherwise, it's not a channel message
170 
171 #if !IGNORE_SYSEX
172  // If we're receiving a SysEx message, it's a SysEx data byte
173  else if (midimsg.header == uint8_t(MIDIMessageType::SYSEX_START)) {
174  // Check if the SysEx buffer has enough space to store the data
175  if (!hasSysExSpace()) {
176  storeByte(midiByte); // Remember to add it next time
177  return MIDIReadEvent::SYSEX_CHUNK;
178  }
179 
180  addSysExByte(midiByte);
181  return MIDIReadEvent::NO_MESSAGE;
182  }
183 #endif // IGNORE_SYSEX
184 
185  // We didn't receive a valid header, so we can't do anything with this data
186  else {
187  DEBUGREF(F("Data byte ignored"));
188  return MIDIReadEvent::NO_MESSAGE;
189  }
