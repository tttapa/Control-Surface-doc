198  uint8_t diff = target.getAddress() - base.getAddress();

215  static bool inRange(uint8_t toMatch, uint8_t base, uint8_t length) {
216  return (base <= toMatch) && (toMatch - base < length);
217  }

228  bool matchBankableAddressInRange(MIDIAddress toMatch,
229  MIDIAddress base) const {
230  using BankableMIDIMatcherHelpers::matchBankable;
231  if (!toMatch.isValid() || !base.isValid())
232  return false;
233  switch (getBankType()) {
234  case CHANGE_ADDRESS: {
235  return toMatch.getChannel() == base.getChannel() &&
236  toMatch.getCableNumber() == base.getCableNumber() &&
237  matchBankableInRange(toMatch.getAddress(),
238  base.getAddress(), length);
239  }
240  case CHANGE_CHANNEL: {
241  return inRange(toMatch.getAddress(), base.getAddress(),
242  length) &&
243  toMatch.getCableNumber() == base.getCableNumber() &&
244  matchBankable(toMatch.getRawChannel(),
245  base.getRawChannel(), getBank());
246  }
247  case CHANGE_CABLENB: {
248  return inRange(toMatch.getAddress(), base.getAddress(),
249  length) &&
250  toMatch.getChannel() == base.getChannel() &&
251  matchBankable(toMatch.getRawCableNumber(),
252  base.getRawCableNumber(), getBank());
253  }
254  default: return false;
255  }
256  }
