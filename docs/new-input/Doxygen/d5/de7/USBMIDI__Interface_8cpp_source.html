112 void USBMIDI_Interface::sendSysExImpl(const SysExMessage msg) {
113  // Don't bother trying to send empty messages
114  if (msg.length == 0)
115  return;
116 
117  Cable cable = msg.cable;
118  uint16_t length = msg.length;
119  const uint8_t *data = msg.data;
120 
121  // Even if the previous chunk wasn't terminated correctly, if this is a new
122  // SysEx message, always forget the previous unsent chunk.
123  if (msg.isFirstChunk()) {
124  // TODO: send a SysExEnd for previous chunk?
125  storedSysExLength = 0;
126  }
127 
128  // Complete the previous unsent chunk to (at most) 3 bytes
129  while (length > 0 && storedSysExLength < 3) {
130  storedSysExData[storedSysExLength++] = *data++;
131  --length;
132  }
133 
134  // If all bytes of the new chunk were used, there are <= 3 stored bytes
135  if (length == 0) {
136  // If this chunk is the last one, terminate the SysEx (termination can
137  // have 1, 2 or 3 bytes, so no need to store anything)
138  if (msg.isLastChunk()) {
139  sendSysExEnd(storedSysExData, storedSysExLength, cable);
140  storedSysExLength = 0;
141  }
142  // If it's the end of the chunk but not the end of the SysEx, and if we
143  // have exactly 3 bytes left, we can send them immediately
144  else if (storedSysExLength == 3) {
145  sendSysExStartCont1(storedSysExData, cable);
146  storedSysExLength = 0;
147  }
148  // If we have less than 3 bytes, we cannot send them now, we have to
149  // store them and wait for the next chunk
150  }
151  // If there are new bytes left in the chunk, there are exactly 3 stored
152  // bytes
153  else {
154  // First send the 3 stored bytes
155  sendSysExStartCont1(storedSysExData, cable);
156  // Then send whatever new data is left in the new chunk (but leave at
157  // least 3 bytes)
158  sendSysExStartCont(data, length, cable);
159  // If this chunk is the last one, terminate the SysEx (termination can
160  // have 1, 2 or 3 bytes, so no need to store anything)
161  if (msg.isLastChunk()) {
162  sendSysExEnd(data, length, cable);
163  storedSysExLength = 0;
164  }
165  // If it's the end of the chunk but not the end of the SysEx, and if we
166  // have exactly 3 bytes left, we can send them immediately
167  else if (length == 3) {
168  sendSysExStartCont1(data, cable);
169  storedSysExLength = 0;
170  }
171  // If we have 1 or 2 bytes left, we cannot send them now, so store them
172  // until we get enough data to fill a 3-byte packet
173  else {
174  storedSysExLength = length;
175  memcpy(storedSysExData, data, length);
176  }
177  }
178  if (alwaysSendImmediately_)
179  sendUSBNow();
180 }
