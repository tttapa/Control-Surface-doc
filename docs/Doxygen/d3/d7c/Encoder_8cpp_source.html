9Encoder::Encoder(uint8_t pin1, uint8_t pin2) : pin1(pin1), pin2(pin2) {
10 // It's much faster to use the GPIO registers directly, rather than
11 // calling digitalRead every time we need to read a pin.
12 // digitalRead looks up the register and bitmasks every time it's called
13 // but here, we look them up once in the constructor.
14 encoder.pin1_register = PIN_TO_BASEREG(pin1);
15 encoder.pin1_bitmask = PIN_TO_BITMASK(pin1);
16 encoder.pin2_register = PIN_TO_BASEREG(pin2);
17 encoder.pin2_bitmask = PIN_TO_BITMASK(pin2);
18 encoder.position = 0;
19}

23Encoder &Encoder::operator=(Encoder &&other) {
24 // First swap the normal member variables:
25 enc_util::swap(this->pin1, other.pin1);
26 enc_util::swap(this->pin2, other.pin2);
27 enc_util::swap(this->interrupts_in_use, other.interrupts_in_use);
28
29 // Next, update the pointers in interruptArgs:
30 // When interrupts are in use, there is a global interrupt context
31 // variable that holds a pointer to the encoders that are being swapped
32 // or moved.
33 // After moving, this pointer would no longer be valid, so it has to be
34 // changed to point to the new encoder object.
35 // Calling attachInterrupt is not necessary, because this should already
36 // have happened in the begin method if interrupts_in_use is nonzero.
37 // Before messing with the state variables that can be changed or
38 // accessed from within an ISR, we have to disable interrupts.
39 noInterrupts();
40 enc_util::swap(this->encoder, other.encoder);
41 if (this->interrupts_in_use > 0) {
42 auto int1 = digitalPinToInterrupt(this->pin1);
43 if (int1 != NOT_AN_INTERRUPT)
44 interruptArgs[int1] = &this->encoder;
45 auto int2 = digitalPinToInterrupt(this->pin2);
46 if (int2 != NOT_AN_INTERRUPT)
47 interruptArgs[int2] = &this->encoder;
48 }
49 if (other.interrupts_in_use > 0) {
50 auto int1 = digitalPinToInterrupt(other.pin1);
51 if (int1 != NOT_AN_INTERRUPT)
52 interruptArgs[int1] = &other.encoder;
53 auto int2 = digitalPinToInterrupt(other.pin2);
54 if (int2 != NOT_AN_INTERRUPT)
55 interruptArgs[int2] = &other.encoder;
56 }
57 interrupts();
58
59 return *this;
60}

69void Encoder::begin() {
70 pinMode(pin1, INPUT_PULLUP);
71 pinMode(pin2, INPUT_PULLUP);
72 // allow time for a passive R-C filter to charge
73 // through the pullup resistors, before reading
74 // the initial state
75 delayMicroseconds(2000);
76 uint8_t s = 0;
77 if (DIRECT_PIN_READ(encoder.pin1_register, encoder.pin1_bitmask))
78 s |= 1;
79 if (DIRECT_PIN_READ(encoder.pin2_register, encoder.pin2_bitmask))
80 s |= 2;
81 encoder.state = s;
82
83 attachInterruptCtx(digitalPinToInterrupt(pin1));
84 attachInterruptCtx(digitalPinToInterrupt(pin2));
85}
